\begin{enumerate}
    \item \textbf{(10 points)} Consider the frequency response
    \[
    H(e^{j\omega}) = 1 + e^{-j\omega}
    \]
     Find the \textbf{real part} of $H(e^{j\omega})$ as a function of $\omega$. \textbf{Note:} don't leave any $j$'s, $i$'s, $\sqrt{-1}$'s, etc. \\
    
    
    % \vspace{8cm}

    \ifthenelse{\boolean{showresults}}{
        \textbf{Solution:}
        
        We can expand the complex exponential using Euler's formula:
        \[
        H(e^{j\omega}) = 1 + e^{-j\omega} = 1 + \cos(-\omega) + j\sin(-\omega)
        \]
        
        Using the even/odd properties of cosine and sine:
        \[
        H(e^{j\omega}) = 1 + \cos(\omega) - j\sin(\omega)
        \]
        
        Therefore, the real part of $H(e^{j\omega})$ is:
        \[
        \boxed{\text{Re}\{H(e^{j\omega})\} = 1 + \cos(\omega)}
        \]
        
    }{}
    \end{enumerate}
